### Task 3: Add i18n Engine + Language Switcher to index.html

**Files:**
- Modify: `frontend/index.html`

**Step 1: Add the i18n engine at the top of the `<script>` block (before `const API = ...`)**

```javascript
// === i18n Engine ===
let _locale = {};
let _currentLang = 'en';

function detectLang() {
  const stored = localStorage.getItem('lang');
  if (stored) return stored;
  const nav = navigator.language || navigator.userLanguage || 'en';
  return nav.startsWith('zh') ? 'zh-TW' : 'en';
}

async function loadLocale(lang) {
  try {
    const resp = await fetch(`${window.location.origin}/locales/${lang}.json`);
    if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
    _locale = await resp.json();
    _currentLang = lang;
  } catch(e) {
    console.warn(`Failed to load locale ${lang}, falling back to en`);
    if (lang !== 'en') {
      return loadLocale('en');
    }
  }
}

function t(key, fallback) {
  const val = _locale[key];
  if (val !== undefined) return val;
  return fallback !== undefined ? fallback : key;
}

function tFormat(key, ...args) {
  let val = t(key);
  args.forEach((arg, i) => { val = val.replace(`{${i}}`, arg); });
  return val;
}

function applyTranslations() {
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.getAttribute('data-i18n');
    const val = _locale[key];
    if (val !== undefined) el.textContent = val;
  });
  // Update lang toggle button states
  document.querySelectorAll('.lang-btn').forEach(btn => {
    btn.classList.toggle('active', btn.dataset.lang === _currentLang);
  });
}

async function setLang(lang) {
  localStorage.setItem('lang', lang);
  await loadLocale(lang);
  applyTranslations();
  // Re-render dynamic content
  const sel = document.getElementById('strategy');
  if (sel && sel.value) renderParams(sel.value);
}
```
